# 求最大公因数(gcb)、最小公倍数(lcm)

/**
 * 最大公因数；辗转相除法
 * @param {*} a 
 * @param {*} b 
 */
function gcb(a, b){
   if(a % b === 0){
      return b;
   }else{
      return gcb(b, a % b);
   }
}

/**
 * 最小公倍数
 * @param {*} a 
 * @param {*} b 
 */
function lcm(a, b){
   return a * b / gcb(a,b);
}

console.log(lcm(6,9));
